setenv bootfile zImage
setenv bootcmd 'run mmcargs; run mmcload; run doboot;'
setenv mmcargs 'setenv bootargs "console=${console} root=/dev/mmcblk0p1 rw rootwait rootfstype=ext4"'
setenv mmcload 'load mmc 0 ${kernel_addr_r} /boot/${bootfile}; load mmc 0 ${fdt_addr_r} /boot/${fdtfile};'
setenv doboot 'bootz ${kernel_addr_r} - ${fdt_addr_r}'
